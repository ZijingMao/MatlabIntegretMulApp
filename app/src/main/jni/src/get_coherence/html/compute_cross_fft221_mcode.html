1function [P] = compute_cross_fft(x, y, win, NFFT)
 2
 3% make it to column
 4x = x(:);
 5y = y(:);
 6
 7% Window the data
 8xw = x.*win;
 9yw = y.*win;
10% compensates for the power of the window.
11U = win'*win;
12
13% fft the data
14[Xx] = fft(xw,NFFT);
15[Yy] = fft(yw,NFFT);
16
17P = Xx.*conj(Yy)/U;      % Auto spectrum.
18
19end

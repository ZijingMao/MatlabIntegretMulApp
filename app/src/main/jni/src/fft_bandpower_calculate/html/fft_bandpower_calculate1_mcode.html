1function [ totalpower, pband ] = fft_bandpower_calculate( data, Fs, band )
 2%fft_bandpower_calculate
 3%   data: the signal of 1 dimension vector
 4%   Fs: the frequency provide
 5%   band: the band provided, should have upper and lower bound
 6
 7if length(band) ~= 2
 8    return;
 9end
10
11data_length = length(data); % get of the data
12
13NFFT = 2^nextpow2(data_length);
14
15X = fft(data,NFFT)/data_length;
16
17freq_space = linspace(0,1,NFFT/2+1);
18
19f = Fs / 2 .* freq_space;
20
21power = 2 * abs(X(1:NFFT/2+1));
22
23% see if band power is in side the frequency range
24if band(2) > f(end)
25    band(2) = f(end);
26end
27
28if band(1) < f(1)   % if the band lower bound is less than f(1)
29    band(1) = f(1);
30end
31
32freq_range = f>=band(1) & f<=band(2);
33
34totalpower = [sum(power .^ 2)];
35
36pband = [sum(power(freq_range) .^ 2)];
37
38end

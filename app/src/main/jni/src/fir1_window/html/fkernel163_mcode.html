1function b = fkernel(m, f, w)
 2    l = length(w);
 3    b = zeros(1, l);
 4    m = -m / 2 : m / 2;
 5    b(m == 0) = 2 * pi * f; % No division by zero
 6    b(m ~= 0) = sin(2 * pi * f * m(m ~= 0)) ./ m(m ~= 0); % Sinc
 7    b = b .* w; % Window
 8    b = b / sum(b); % Normalization to unity gain at DC
 9    
10end

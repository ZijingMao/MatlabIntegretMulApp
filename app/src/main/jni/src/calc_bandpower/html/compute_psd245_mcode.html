1function [ Pxx, w ] = compute_psd(Sxx,w,nfft,Fs)
 2
 3select = 1:nfft/2+1;    % EVEN
 4Sxx_unscaled = Sxx(select,:); % Take only [0,pi] or [0,pi)
 5% Don't double unique Nyquist point
 6Sxx = [Sxx_unscaled(1,:); 2*Sxx_unscaled(2:end-1,:); Sxx_unscaled(end,:)]; 
 7w = w(select);
 8
 9Pxx = Sxx./Fs; % Scale by the sampling frequency to obtain the psd

11end
